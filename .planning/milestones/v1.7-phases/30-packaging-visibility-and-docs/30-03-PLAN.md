function extractTable(content, heading, filePath) {
  const start = content.indexOf(heading);
  if (start === -1) {
    throw new Error(`${filePath}: missing heading ${heading}`);
  }
  const lines = content.slice(start + heading.length).split(/\r?\n/);
  const table = [];
  let inTable = false;
  for (const line of lines) {
    if (/^\|/.test(line)) {
      inTable = true;
      table.push(line);
      continue;
    }
    if (inTable) {
      break;
    }
  }
  if (table.length === 0 || table.length === 1 || table.length === 2) {
    throw new Error(`${filePath}: traceability table is missing or incomplete`);
  }
  return table;
}

function parseTable(table, filePath) {
  const rows = table.map((line) => line.trim().slice(1, -1).split('|').map((cell) => cell.trim()));
  const header = rows[0];
  const requirementIndex = header.indexOf('Requirement');
  const phaseIndex = header.indexOf('Phase');
  if (requirementIndex === -1 || phaseIndex === -1) {
    throw new Error(`${filePath}: table must include Requirement and Phase columns`);
  }
  return rows.slice(2).map((cells) => ({
    requirement: cells[requirementIndex],
    phase: cells[phaseIndex],
  }));
}

for (const spec of specs) {

    seen.add(row.requirement);
  }
